代码2：代码编写

		var myradio = new hait.form.field.RadioGroupField();
		myradio.setParam({
			defaultValue : "id",
			options : [ {
				text : "编号",
				val : "id"
			}, {
				text : "名称",
				val : "name"
			} ]
		});
		myradio.draw();
		$("#myradio1").append(myradio.obj);
